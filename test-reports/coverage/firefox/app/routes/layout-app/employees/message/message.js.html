        const id = $stateParams.id;
        $modal.open({
            template: template,
            resolve: {
                init: ['EmployeeModel', EmployeeModel => EmployeeModel.initItem(id)]
            },
            controller: EmployeeMessage,
            controllerAs: 'vm',
            size: 'md'
        }).result.finally(ModalService.onFinal('app.employees'));
    }]
})
